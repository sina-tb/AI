lived_pdf = stats.norm.pdf(x,lived_mean,lived_std)
dead_pdf = stats.norm.pdf(x,dead_mean,dead_std)

intersections = []
prev_dif = 0
t0, prev_c1, prev_c2 = None, None, None
for t1, c1, c2 in zip(x, lived_pdf, dead_pdf):
    new_dif = c2 - c1
    if new_dif * prev_dif < 0:
        denom = prev_dif - new_dif
        intersections.append((-new_dif*t0  + prev_dif*t1) / denom)
    t0, prev_c1, prev_c2, prev_dif = t1, c1, c2, new_dif
print(intersections)
